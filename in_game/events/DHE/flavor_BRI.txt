flavor_bri.2 = {
	type = country_event
	title = flavor_bri.2.title
	desc = flavor_bri.2.desc
	fire_only_once = yes
	
	dynamic_historical_event = {
		tag = BRI
		from = 1340.1.1
		to = 1365.1.1
		monthly_chance = 10
	}

	illustration_tags = {
        10 = angry
        10 = interior
    }

	trigger = {
		at_war = no
		owns = location:le_faou
	}

	immediate = {
		event_illustration_estate_effect = { foreground = estate_type:clergy_estate background = estate_type:nobles_estate }
		
		location:le_faou = {
			save_scope_as = target_location
		}

		scope:target_location = { create_dynasty_from_location = de_montfort_dynasty }
		create_character = {
			first_name = name_john
			dynasty = dynasty:de_montfort_dynasty
			birth_date = 1295.5.13
			birth_location = location:le_faou
			adm = { 66 100 }
			dip = { 50 100 }
			mil = { 50 100 }
			script = john_rockfor_script
			save_scope_as = target_character
			add_trait = trait:efficient_administrator
			estate = estate_type:nobles_estate
		}

		if = {
			limit = {
				any_character = {
					is_female = yes
					is_married = no
					is_adult = yes
				}
			}
			random_character = {
				limit = {
					is_female = yes
					is_married = no
					is_adult = yes
				}
				save_scope_as = target_lady
			}
		}
		else = {
			create_character = {
				age = 28
				female = yes
				save_scope_as = target_lady
			}
		}
	}
	
	option = {
		name = flavor_bri.2.a
		
		scope:target_lady = { 
			marry_character = scope:target_character 
		}
		scope:target_character = { impregnate = scope:target_lady }
	}
}

flavor_bri.3 = {
	type = country_event
	title = flavor_bri.3.title
	desc = flavor_bri.3.desc
	fire_only_once = yes
	
	dynamic_historical_event = {
		tag = BRI
		from = 1340.1.1
		to = 1365.1.1
		monthly_chance = 30
	}
	
	illustration_tags = {
		10 = angry
        10 = interior
    }
	
	trigger = { 
		dynasty_exists = de_montfort_dynasty
		character:john_rockfor_script = {
			is_alive = yes
			owner = root
		}
		any_character = {
			OR = {
				this = character:john_rockfor_script
				any_parent = { this = character:john_rockfor_script }
			}
		}
	}

	immediate = {
		event_illustration_estate_effect = { foreground = estate_type:clergy_estate background = estate_type:nobles_estate }

		random_character = {
			limit = {
				OR = {
					this = character:john_rockfor_script
					any_parent = { this = character:john_rockfor_script }
				}
			}
			save_scope_as = target_character
		}

	}
	
	option = {
		name = flavor_bri.3.a
		
		make_subject_of = {
			target = c:ENG
			type = subject_type:vassal
		}
		
		set_new_ruler = scope:target_character
		
		add_legitimacy = legitimacy_mild_penalty
		ai_chance = {
			factor = 7
		}
	}
	
	option = {
		name = flavor_bri.3.b
		
		add_stability = stability_ultimate_penalty
		#set_variable = { name = zel_john_de_montfort value = 0 }
		
		ai_chance = {
			factor = 3
		}
	}
}
